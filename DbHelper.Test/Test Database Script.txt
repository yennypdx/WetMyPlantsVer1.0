Here is the script that I used to set up the test database.
This database must be present and your local SQL Server Agent must be running for unit tests to pass.

CREATE DATABASE WetMyPlantsTest;
GO

USE WetMyPlantsTest;
GO

CREATE TABLE Users (
	UserID			INT IDENTITY PRIMARY KEY,
	FirstName		nvarchar(32),
	LastName		nvarchar(32),
	Hash			nvarchar(MAX),
	Phone			nvarchar(18),
	Email			nvarchar(100)
);
GO

CREATE TABLE Tokens (
	ID				INT IDENTITY PRIMARY KEY,
	UserID			INT,
	Token			nvarchar(MAX),
	Expiry			DATETIME
);
GO

SELECT * FROM Users;
SELECT * FROM Tokens;
